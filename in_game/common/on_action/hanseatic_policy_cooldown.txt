on_international_organization_policy_changed = {
    # Apply a temporary cooldown to the leader city when a Hanseatic policy changes
    limit = { international_organization = international_organization:hanseatic_league }
    any_country = {
        limit = { is_leader_of_international_organization = international_organization:hanseatic_league }
        add_country_modifier = { modifier = hl_policy_proposal_cooldown years = 10 }
    }
}
