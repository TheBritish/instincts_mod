hl_trade_regulations = {
    type = international_organization
    law_category = socioeconomic
    potential = { international_organization_type = international_organization_type:hanseatic_league }

    hl_strict_regulations = {
        wants_propose_policy = {
            add = { desc = "POLICY_LEADER_PREFERENCE" value = 100 }
            add = { desc = "POLICY_RECENTLY_PROPOSED" value = -80 if = { limit = { has_country_modifier = hl_policy_proposal_cooldown } } }
        }
        wants_this_policy_bias = {
            add = { desc = "POLICY_DEFAULT" value = 50 }
        }
        # Strict regulations centralize control: IO coordination raises merchant capacity.
        # The league leader gains tariff income and merchant capacity; members suffer lower
        # trade efficiency and greater local unrest.
        international_organization_modifier = { trade_efficiency = 0.04 global_merchant_capacity_modifier = 0.10 }
        # Leader-specific benefits (tariff income, merchant capacity)
        country_modifier = {
            potential_trigger = {
                is_leader_of_international_organization = international_organization:hanseatic_league
            }
            monthly_gold_income = 0.02
            global_merchant_capacity_modifier = 0.05
        }
        # Effects applied to member countries
        country_modifier = { trade_efficiency = -0.05 local_unrest = 0.10 }
    }

    hl_moderate_regulations = {
        wants_propose_policy = {
            add = { desc = "POLICY_LEADER_PREFERENCE" value = 50 }
            add = { desc = "POLICY_RECENTLY_PROPOSED" value = -60 if = { limit = { has_country_modifier = hl_policy_proposal_cooldown } } }
        }
        wants_this_policy_bias = {
            add = { desc = "POLICY_DEFAULT" value = 25 }
        }
        # Moderate regulations: modest IO trade benefits. Leader receives small tariff income
        # and merchant capacity boost; members receive a small trade penalty.
        international_organization_modifier = { trade_efficiency = 0.03 }
        country_modifier = {
            potential_trigger = {
                is_leader_of_international_organization = international_organization:hanseatic_league
            }
            monthly_gold_income = 0.01
            global_merchant_capacity_modifier = 0.05
        }
        country_modifier = { trade_efficiency = -0.01 }
    }

    hl_minimal_regulations = {
