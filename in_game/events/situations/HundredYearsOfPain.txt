        add_estate_satisfaction = { estate = burghers satisfaction = -20 }
		c:ENG = {
			create_relation = {
				first = c:ENG
				second = root
				type = relation_type:embargo_nation
			}
		}

		kill_character = {
			target = scope:target_character2
			reason = execution
		}
	}
