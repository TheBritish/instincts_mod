namespace = flavor_burgundy

# Phase I - The Golden Age
country_event = {
	id = flavor_burgundy.1
	title = flavor_burgundy.1.t
	desc = flavor_burgundy.1.d
	picture = MILITARY_COURT_scene
	fire_only_once = yes

	trigger = {
		tag = BUR
		OR = {
			current_date >= 1380.1.1
			ruler = { has_personality = militarist_personality }
		}
		OR = {
			any_owned_province = { region = region:low_countries_region }
			any_subject_country = { any_owned_province = { region = region:low_countries_region } }
		}
	}

	option = {
		name = flavor_burgundy.1.a
		add_country_modifier = { name = middle_kingdom_diplomacy duration = 18250 }
		if = {
			limit = { FRA = { exists = yes is_rival = yes } }
			add_historical_friend = ENG
		}
	}
}

# Phase II - The Catastrophe of 1477
country_event = {
	id = flavor_burgundy.10
	title = flavor_burgundy.10.t
	desc = flavor_burgundy.10.d
	picture = DYNASTY_scene
	fire_only_once = yes

	trigger = {
		tag = BUR
		is_at_war = yes
		OR = {
			any_war = { limit = { is_current = yes } war_score < -25 }
			has_country_flag = burgundy_duke_died_in_battle
		}
	}

	immediate = {
		if = {
			limit = { FRA = { exists = yes } }
			every_owned_province = {
				limit = { region = region:france_region }
				cede_province = FRA
			}
		}
		# Move capital to a northern Low Countries province
		random_owned_province = {
			limit = { region = region:low_countries_region is_capital = no }
			set_capital = THIS
		}
	}

	option = {
		name = flavor_burgundy.10.a # The Habsburg Marriage (Historical)
		if = {
			limit = { HAB = { exists = yes } }
			become_subject = { type = personal_union overlord = HAB }
		}
		add_country_modifier = { name = imperial_protection duration = 3650 }
	}

	option = {
		name = flavor_burgundy.10.b # The Duchess Stands Alone
		add_opinion = { who = FRA value = -150 }
		add_opinion = { who = HAB value = -150 }
		every_owned_province = {
			limit = { region = region:low_countries_region }
			spawn_rebels = { type = dutch_patriots size = 2 }
		}
	}
}

# Phase III - The Burgundian Circle
country_event = {
	id = flavor_burgundy.20
	title = flavor_burgundy.20.t
	desc = flavor_burgundy.20.d
	picture = ADMINISTRATION_scene
	fire_only_once = yes

	trigger = {
		tag = BUR
		current_date >= 1500.1.1
		any_owned_province = { region = region:low_countries_region }
	}

	option = {
		name = flavor_burgundy.20.a
		# Integrate remaining vassals in the Low Countries
		every_subject_country = {
			limit = { any_owned_province = { region = region:low_countries_region } }
			inherit = ROOT
		}
		add_country_modifier = { name = burgundy_pragmatic_sanction duration = -1 }
		every_owned_province = {
			limit = { region = region:low_countries_region }
			add_province_modifier = { name = burgundian_circle duration = -1 }
		}
	}
}
